  <script>
    // Variáveis globais
    let scene, camera, renderer;
    let bible, coverLeft, coverRight, pages, spine;
    let isBibleOpen = false;
    let isRotating = false;

    // Versículos para display

    // Inicialização
    function init() {
      // Cena
      scene = new THREE.Scene();
      scene.fog = new THREE.FogExp2(0x1a1a2e, 0.02);

      // Câmera
      camera = new THREE.PerspectiveCamera(45, 600 / 400, 0.1, 1000);
      camera.position.set(0, 3, 8);
      camera.lookAt(0, 0, 0);

      // Renderizador
      const container = document.getElementById('canvas-container');
      renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
      renderer.setSize(600, 400);
      renderer.setPixelRatio(window.devicePixelRatio);
      renderer.shadowMap.enabled = true;
      renderer.shadowMap.type = THREE.PCFSoftShadowMap;
      container.appendChild(renderer.domElement);

      // Iluminação
      const ambientLight = new THREE.AmbientLight(0x404040, 0.5);
      scene.add(ambientLight);

      const mainLight = new THREE.PointLight(0xffd700, 1.2, 20);
      mainLight.position.set(5, 5, 5);
      mainLight.castShadow = true;
      mainLight.shadow.mapSize.width = 2048;
      mainLight.shadow.mapSize.height = 2048;
      scene.add(mainLight);

      const fillLight = new THREE.PointLight(0x4a90e2, 0.4, 15);
      fillLight.position.set(-5, 2, -5);
      scene.add(fillLight);

      const rimLight = new THREE.SpotLight(0xffffff, 0.6);
      rimLight.position.set(0, 10, -5);
      rimLight.lookAt(0, 0, 0);
      scene.add(rimLight);

      // Criar Bíblia
      createBible();

      // Adicionar partículas de luz
      createLightParticles();

      // Animação inicial
      gsap.from(bible.position, { y: -5, duration: 1.5, ease: "power2.out" });
      gsap.from(bible.rotation, { x: Math.PI/4, duration: 2, ease: "power2.out" });

      // Loop de renderização
      animate();

      // Responsivo
      window.addEventListener('resize', onWindowResize);
    }

    function createBible() {
      bible = new THREE.Group();

      // Materiais
      const coverMaterial = new THREE.MeshStandardMaterial({
        color: 0x8B4513, // Marrom couro
        roughness: 0.7,
        metalness: 0.1,
        bumpScale: 0.02
      });

      const pageMaterial = new THREE.MeshStandardMaterial({
        color: 0xfff8dc, // Cor de papel envelhecido
        roughness: 0.9,
        metalness: 0.0,
        side: THREE.DoubleSide
      });

      const spineMaterial = new THREE.MeshStandardMaterial({
        color: 0x654321,
        roughness: 0.8,
        metalness: 0.2
      });

      const goldMaterial = new THREE.MeshStandardMaterial({
        color: 0xd4af37,
        roughness: 0.3,
        metalness: 0.8
      });

      // Dimensões
      const width = 2.5;
      const height = 3.5;
      const thickness = 0.5;
      const pageThickness = 0.01;

      // Lombada (Spine)
      spine = new THREE.Mesh(
        new THREE.BoxGeometry(0.15, height, thickness),
        spineMaterial
      );
      spine.position.x = -width / 2;
      spine.castShadow = true;
      bible.add(spine);

      // Capas (Cover) - usando múltiplos planos para simular páginas
      coverLeft = new THREE.Group();
      coverRight = new THREE.Group();

      // Criar páginas empilhadas
      const pageCount = 30;
      for (let i = 0; i < pageCount; i++) {
        const offset = (i - pageCount/2) * pageThickness;
        
        // Página esquerda
        const pageL = new THREE.Mesh(
          new THREE.BoxGeometry(width/2 - 0.05, height - 0.1, pageThickness),
          pageMaterial
        );
        pageL.position.set(-width/4 - 0.05, offset, 0);
        pageL.castShadow = true;
        coverLeft.add(pageL);

        // Página direita
        const pageR = new THREE.Mesh(
          new THREE.BoxGeometry(width/2 - 0.05, height - 0.1, pageThickness),
          pageMaterial
        );
        pageR.position.set(width/4 + 0.05, offset, 0);
        pageR.castShadow = true;
        coverRight.add(pageR);
      }

      // Capas externas
      const coverL = new THREE.Mesh(
        new THREE.BoxGeometry(width/2, height, thickness/4),
        coverMaterial
      );
      coverL.position.set(-width/4 - 0.07, 0, 0);
      coverL.castShadow = true;
      coverLeft.add(coverL);

      const coverR = new THREE.Mesh(
        new THREE.BoxGeometry(width/2, height, thickness/4),
        coverMaterial
      );
      coverR.position.set(width/4 + 0.07, 0, 0);
      coverR.castShadow = true;
      coverRight.add(coverR);

      // Detalhe dourado na capa
      const cross = new THREE.Mesh(
        new THREE.BoxGeometry(0.3, 0.8, 0.02),
        goldMaterial
      );
      cross.position.set(width/4 + 0.08, 0.5, 0);
      coverRight.add(cross);

      const crossH = new THREE.Mesh(
        new THREE.BoxGeometry(0.5, 0.15, 0.02),
        goldMaterial
      );
      crossH.position.set(width/4 + 0.08, 0.7, 0);
      coverRight.add(crossH);

      // Pivot points para animação de abertura
      coverLeft.position.x = -0.05;
      coverRight.position.x = 0.05;

      bible.add(coverLeft);
      bible.add(coverRight);

      // Texto decorativo na lombada
      const title = new THREE.Mesh(
        new THREE.BoxGeometry(0.02, 1.5, thickness + 0.05),
        goldMaterial
      );
      title.position.set(-width/2 - 0.08, 0, 0);
      bible.add(title);

      // Ajustar posição inicial
      bible.position.y = -1;
      
      scene.add(bible);
    }

    function createLightParticles() {
      const particleCount = 50;
      const geometry = new THREE.BufferGeometry();
      const positions = new Float32Array(particleCount * 3);

      for (let i = 0; i < particleCount; i++) {
        positions[i * 3] = (Math.random() - 0.5) * 10;
        positions[i * 3 + 1] = Math.random() * 5;
        positions[i * 3 + 2] = (Math.random() - 0.5) * 10;
      }

      geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));

      const material = new THREE.PointsMaterial({
        color: 0xd4af37,
        size: 0.05,
        transparent: true,
        opacity: 0.6
      });

      const particles = new THREE.Points(geometry, material);
      scene.add(particles);

      // Animar partículas
      gsap.to(particles.rotation, {
        y: Math.PI * 2,
        duration: 20,
        repeat: -1,
        ease: "none"
      });
    }

    // Animações
    function openBible() {
      if (isBibleOpen) return;
      isBibleOpen = true;

      gsap.to(coverLeft.rotation, {
        y: Math.PI / 3,
        duration: 1.5,
        ease: "power2.inOut"
      });

      gsap.to(coverRight.rotation, {
        y: -Math.PI / 3,
        duration: 1.5,
        ease: "power2.inOut"
      });

      gsap.to(bible.position, {
        y: -0.5,
        duration: 1,
        ease: "power2.out"
      });

      document.getElementById('verseDisplay').style.borderColor = '#d4af37';
      document.getElementById('verseDisplay').style.background = 'rgba(212, 175, 55, 0.2)';
    }

    function closeBible() {
      if (!isBibleOpen) return;
      isBibleOpen = false;

      gsap.to(coverLeft.rotation, {
        y: 0,
        duration: 1.5,
        ease: "power2.inOut"
      });

      gsap.to(coverRight.rotation, {
        y: 0,
        duration: 1.5,
        ease: "power2.inOut"
      });

      gsap.to(bible.position, {
        y: -1,
        duration: 1,
        ease: "power2.out"
      });

      document.getElementById('verseDisplay').style.borderColor = 'rgba(212, 175, 55, 0.3)';
      document.getElementById('verseDisplay').style.background = 'rgba(212, 175, 55, 0.1)';
    }

    function rotateBible() {
      if (isRotating) return;
      isRotating = true;

      gsap.to(bible.rotation, {
        y: Math.PI * 2,
        duration: 3,
        ease: "power2.inOut",
        onComplete: () => {
          bible.rotation.y = 0;
          isRotating = false;
        }
      });
    }

    function showVerse() {
      const display = document.getElementById('verseDisplay');
      
      // Fade out
      gsap.to(display, {
        opacity: 0,
        duration: 0.3,
        onComplete: () => {
          verseIndex = (verseIndex + 1) % verses.length;
          display.textContent = `"${verses[verseIndex]}"`;
          
          // Fade in
          gsap.to(display, {
            opacity: 1,
            duration: 0.5
          });
        }
      });

      // Efeito de brilho
      gsap.fromTo(display, 
        { boxShadow: '0 0 0px rgba(212, 175, 55, 0)' },
        { boxShadow: '0 0 30px rgba(212, 175, 55, 0.5)', duration: 0.5, yoyo: true, repeat: 1 }
      );
    }

    function onWindowResize() {
      const container = document.getElementById('canvas-container');
      const width = container.clientWidth;
      const height = container.clientHeight;
      
      camera.aspect = width / height;
      camera.updateProjectionMatrix();
      renderer.setSize(width, height);
    }

    function animate() {
      requestAnimationFrame(animate);
      
      // Rotação suave contínua
      if (!isRotating && !isBibleOpen) {
        bible.rotation.y += 0.002;
      }
      
      renderer.render(scene, camera);
    }

    // Criar partículas HTML decorativas
    function createHTMLParticles() {
      const container = document.getElementById('particles');
      for (let i = 0; i < 20; i++) {
        const particle = document.createElement('div');
        particle.className = 'particle';
        particle.style.left = Math.random() * 100 + '%';
        particle.style.width = Math.random() * 5 + 2 + 'px';
        particle.style.height = particle.style.width;
        particle.style.animationDelay = Math.random() * 10 + 's';
        particle.style.animationDuration = (Math.random() * 10 + 10) + 's';
        container.appendChild(particle);
      }
    }

    // Iniciar quando a página carregar
    window.addEventListener('DOMContentLoaded', () => {
      init();
      createHTMLParticles();
    });
  </script>
</body>
